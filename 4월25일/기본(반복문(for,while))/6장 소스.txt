void main() {
	int i, j,a;
	//상단
	printf("몇개:");
	scanf("%d", &a);
	for (i = 0; i < a; i++) {			
		for (j = i+1; j < a; j++) {
			printf(" ");
		}
		for (j = 0; j <= i*2; j++){
			printf("*");
		}
		printf("\n");
	}
	//하단
	for (i = 0; i < a-1; i++) {		
		for (j = 0; j <= i; j++) {
			printf(" ");
		}
		for (j = 0; j < (a-1-i)*2-1; j++) {
			printf("*");
		}
		printf("\n");
	}
}
